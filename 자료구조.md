    public ListNode reverseList(ListNode head) {
        
        ListNode res = null;

        while(head != null){
            ListNode tmp = new ListNode(head.val);
            tmp.next = res;
            res = tmp;
            head = head.next;
        }

        return res; 
    }
  }

    class Solution {
       public ListNode reverseList(ListNode head) {

              if(head == null || head.next == null){
                  return head;
              }

              ListNode reverseNode = reverseList(head.next);

              head.next.next = head;
              head.next = null;

              return reverseNode;

       }
   }
